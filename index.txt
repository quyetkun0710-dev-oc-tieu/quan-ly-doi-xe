const fd = new FormData(form);
fd.append("tag", tag);
console.group("SUBMIT:", tag);
for (const [k, v] of fd.entries()) {
console.log(k, "=", v);
}
console.groupEnd();
fetch(SCRIPT_URL, {
method: "POST",
body: fd
})
.then(res => res.text())
.then(txt => {
console.log("SERVER:", txt);
let r;
try {
r = JSON.parse(txt);
} catch {
r = {};
}
if (r.status === "success") {
alert("✅ Lưu thành công");
setTimeout(() => form.reset(), 300);
} else {
alert("❌ Lỗi lưu dữ liệu");
}
})
.catch(err => {
console.error(err);
alert("❌ Lỗi kết nối server");
});
}
/* ================= SEARCH ================= */
function searchByKey(tag, keyName, keyValue, formId) {
if (!keyValue) {
alert("⚠️ Thiếu dữ liệu tra cứu");
return;
}
const url =
`${SCRIPT_URL}?tag=${encodeURIComponent(tag)}&${encodeURIComponent(
keyName
)}=${encodeURIComponent(keyValue)}`;
fetch(url)
.then(res => res.json())
.then(data => {
if (!data || Object.keys(data).length === 0) {
alert("❌ Không tìm thấy dữ liệu");
return;
}
fillForm(formId, data);
})
.catch(err => {
console.error(err);
alert("❌ Lỗi tra cứu");
});
}
/* ================= FORM UTIL ================= */
function fillForm(formId, data) {
const form = document.getElementById(formId);
// input thường
Object.keys(data).forEach(k => {
const el = form.querySelector(`[name="${k}"]`);
if (!el || el.type === "radio") return;
el.value = data[k] ?? "";
});
// radio – xử lý RIÊNG
Object.keys(data).forEach(k => {
const radios = form.querySelectorAll(
`input[type="radio"][name="${k}"]`
);
if (!radios.length) return;
radios.forEach(r => {
r.checked = String(r.value) === String(data[k]);
});
});
}

/* ===== TÍNH SỐ NGÀY CÒN LẠI ===== */
function calcRemainDays(dateStr) {
if (!dateStr) return "";
const today = new Date();
today.setHours(0, 0, 0, 0);
const target = new Date(dateStr);
target.setHours(0, 0, 0, 0);
const diff = Math.ceil(
(target - today) / (1000 * 60 * 60 * 24)
);
return diff >= 0 ? diff : 0;
}
function calc(input, limitDays, targetField) {
if (!input || !input.value) return;
const form = input.form;
if (!form || !form[targetField]) return;
const today = new Date();
today.setHours(0, 0, 0, 0);
const target = new Date(input.value);
if (isNaN(target.getTime())) return;
target.setHours(0, 0, 0, 0);
let diff = Math.ceil(
(target - today) / (1000 * 60 * 60 * 24)
);
if (diff < 0) diff = 0;
form[targetField].value = diff;
}

/* ================= AUTO CALC ================= */
function calcNhatKy() {
const f = document.getElementById("nhatkyForm");
if (!f) return;
// ===== TÍNH GIỜ =====
if (f.gio_xuat.value && f.gio_ve.value) {
const x =
new Date("1970-01-01T" + f.gio_xuat.value);
const v =
new Date("1970-01-01T" + f.gio_ve.value);
const diff = (v - x) / 3600000;
f.tong_gio.value = diff > 0 ? diff.toFixed(2) : "";
} else {
f.tong_gio.value = "";
}
// ===== TÍNH KM =====
const kmx = parseFloat(f.km_xuat.value) || 0;
const kmv = parseFloat(f.km_ve.value) || 0;
if (kmv >= kmx && kmx > 0) {
f.tong_km.value = kmv - kmx;
} else {
f.tong_km.value = "";
}
}
function calcTheTichThung() {
const f = document.getElementById("xeForm");
if (!f) return;
const dai = parseFloat(f.thung_dai.value) || 0;
const rong = parseFloat(f.thung_rong.value) || 0;
const cao = parseFloat(f.thung_cao.value) || 0;
if (dai > 0 && rong > 0 && cao > 0) {
f.kich_thuoc_thung.value =
`${dai} x ${rong} x ${cao}`;
f.the_tich_thung.value =
(dai * rong * cao).toFixed(2);
} else {
f.kich_thuoc_thung.value = "";
f.the_tich_thung.value = "";
}
}
function calcChinhThuc() {
const f = document.getElementById("taixeForm");
if (!f) return;
const ngayNhan = f.ngay_nhan_viec.value;
const soThang = parseInt(f.thu_viec.value, 10);
if (!ngayNhan || isNaN(soThang)) {
f.chinh_thuc.value = "";
return;
}
const d = new Date(ngayNhan);
if (isNaN(d.getTime())) {
f.chinh_thuc.value = "";
return;
}
d.setMonth(d.getMonth() + soThang);
const dd = String(d.getDate()).padStart(2, "0");
const mm = String(d.getMonth() + 1).padStart(2, "0");
const yyyy = d.getFullYear();
f.chinh_thuc.value = `${dd}/${mm}/${yyyy}`;
}
function loadTableByKey(tag, keyName, keyValue, tableId) {
const table = document.getElementById(tableId);
if (!table) return;
const thead = table.querySelector("thead");
const tbody = table.querySelector("tbody");
thead.innerHTML = "";
tbody.innerHTML = "";
/* ✅ KHÔNG CÓ KEY → LẤY HẾT */
let url = `${SCRIPT_URL}?tag=${encodeURIComponent(tag)}`;
/* ✅ CHỈ GẮN KEY KHI CÓ */
if (keyValue) {
url += `&${encodeURIComponent(keyName)}=${encodeURIComponent(keyValue)}`;
}
fetch(url)
.then(res => res.json())
.then(list => {
if (!Array.isArray(list) || list.length === 0) {
tbody.innerHTML = `
<tr>
<td colspan="100%" class="has-text-centered">
Không có dữ liệu
</td>
</tr>
`;
return;
}
const headers = Object.keys(list[0]);
/* ===== HEADER ===== */
thead.innerHTML =
"<tr><th>Chọn</th>" +
headers.map(h => `<th>${h}</th>`).join("") +
"</tr>";
/* ===== BODY ===== */
list.forEach(row => {
const tr = document.createElement("tr");
/* === CỘT CHỌN === */
const tdSelect = document.createElement("td");
const btn = document.createElement("button");
btn.textContent = "Chọn";
btn.className = "button is-small is-link";
btn.onclick = () => {
if (tag === "xe") fillForm("xeForm", row);
if (tag === "taixe") fillForm("taixeForm", row);
};
tdSelect.appendChild(btn);
tr.appendChild(tdSelect);
/* === CÁC CỘT DỮ LIỆU === */
headers.forEach(h => {
const td = document.createElement("td");
td.textContent = row[h] ?? "";
tr.appendChild(td);
});
tbody.appendChild(tr);
});
})
.catch(() => {
tbody.innerHTML = `
<tr>
<td colspan="100%" class="has-text-danger">
Lỗi tải dữ liệu
</td>
</tr>
`;
});
}
/* ================= DATE UTIL ================= */
function calcRemainDays(dateStr) {
if (!dateStr) return "";
const d = new Date(dateStr);
if (isNaN(d)) return "";
const today = new Date();
today.setHours(0, 0, 0, 0);
d.setHours(0, 0, 0, 0);
return Math.round((d - today) / 86400000);
}
function calc(el, soNgay, targetField) {
const f = el.closest("form");
if (!f) return;
if (!el.value) {
if (f.elements[targetField]) f.elements[targetField].value = "";
return;
}
const ngayHetHan = new Date(el.value);
ngayHetHan.setDate(ngayHetHan.getDate() + soNgay);
const today = new Date();
today.setHours(0, 0, 0, 0);
ngayHetHan.setHours(0, 0, 0, 0);
const conLai = Math.round((ngayHetHan - today) / 86400000);
if (f.elements[targetField]) {
f.elements[targetField].value = conLai;
}
}

}

/* ================= NHIÊN LIỆU ================= */
function toNumber(v) {
if (!v) return 0;
return (
Number(
String(v)
.replace(/\./g, "")
.replace(/,/g, ".")
) || 0
);
}
function calcFuelKM() {
const f = document.getElementById("fuelForm");

kmSau >= kmTruoc && kmTruoc > 0 ? kmSau - kmTruoc : "";
}

function loadNhatKyTableByKey(tag, keyName, keyValue, tableId) {

thead.innerHTML = "";
tbody.innerHTML = "";
const url =
`${SCRIPT_URL}?tag=${encodeURIComponent(tag)}` +
`&${encodeURIComponent(keyName)}=${encodeURIComponent(keyValue)}`;
fetch(url)
.then(res => res.json())
.then(rows => {
if (!rows || !rows.length) {
alert("❌ Không có dữ liệu");
return;
}
const headers = Object.keys(rows[0]);
thead.innerHTML =
`<tr>${headers.map(h => `<th>${h}</th>`).join("")}</tr>`;
tbody.innerHTML =
rows
.map(r =>
`<tr>${headers.map(h => `<td>${r[h] ?? ""}</td>`).join("")}</tr>`
)
.join("");
})
.catch(err => {
console.error(err);
alert("❌ Lỗi tải dữ liệu nhật ký");
});
}
